70 ILboolean iIsValidMdl(void)
71 {
72  ILuint Id, Version;
73 
74  Id = GetLittleUInt();
75  Version = GetLittleUInt();
76  iseek(-8, IL_SEEK_CUR); // Restore to previous position.
77

80  return IL_FALSE;
81  return IL_TRUE;

127 ILboolean iLoadMdlInternal()
128 {
129  ILuint Id, Version, NumTex, TexOff, TexDataOff, Position, ImageNum;
130  ILubyte *TempPal;
131  TEX_HEAD TexHead;
132  ILimage *BaseImage=NULL;
133  ILboolean BaseCreated = IL_FALSE;
134 
135  if (iCurImage == NULL) {
136  ilSetError(IL_ILLEGAL_OPERATION);
137  return IL_FALSE;
138  }
139 
140  Id = GetLittleUInt();
141  Version = GetLittleUInt();
142 
143  // 0x54534449 == "IDST"
144  if (Id != 0x54534449 || Version != 10) {
145  ilSetError(IL_INVALID_FILE_HEADER);
146  return IL_FALSE;
147  }
148 
149  // Skips the actual model header.
150  iseek(172, IL_SEEK_CUR);
151 
152  NumTex = GetLittleUInt();
153  TexOff = GetLittleUInt();
154  TexDataOff = GetLittleUInt();
155 
156  if (NumTex == 0 || TexOff == 0 || TexDataOff == 0) {
157  ilSetError(IL_ILLEGAL_FILE_VALUE);
158  return IL_FALSE;
159  }
160 
161  iseek(TexOff, IL_SEEK_SET);
162 
163  for (ImageNum = 0; ImageNum < NumTex; ImageNum++) {
164  if (iread(TexHead.Name, 1, 64) != 64)
165  return IL_FALSE;
166  TexHead.Flags = GetLittleUInt();
167  TexHead.Width = GetLittleUInt();
168  TexHead.Height = GetLittleUInt();
169  TexHead.Offset = GetLittleUInt();
170  Position = itell();
171 
172  if (TexHead.Offset == 0) {
173  ilSetError(IL_ILLEGAL_FILE_VALUE);
174  return IL_FALSE;
175  }
176 
177  if (!BaseCreated) {
178  ilTexImage(TexHead.Width, TexHead.Height, 1, 1, IL_COLOUR_INDEX, IL_UNSIGNED_BYTE, NULL);
179  iCurImage->Origin = IL_ORIGIN_LOWER_LEFT;
180  BaseCreated = IL_TRUE;
181  BaseImage = iCurImage;
182  //iCurImage->NumNext = NumTex - 1; // Don't count the first image.
183  }
184  else {
185  //iCurImage->Next = ilNewImage(TexHead.Width, TexHead.Height, 1, 1, 1);
186  iCurImage = iCurImage->Next;
187  iCurImage->Format = IL_COLOUR_INDEX;
188  iCurImage->Type = IL_UNSIGNED_BYTE;
189  }
190 
191  TempPal = (ILubyte*)ialloc(768);
192  if (TempPal == NULL) {
193  iCurImage = BaseImage;
194  return IL_FALSE;
195  }
196  iCurImage->Pal.Palette = TempPal;
197  iCurImage->Pal.PalSize = 768;
198  iCurImage->Pal.PalType = IL_PAL_RGB24;
199 
200  iseek(TexHead.Offset, IL_SEEK_SET);
201  if (iread(iCurImage->Data, TexHead.Width * TexHead.Height, 1) != 1)
202  return IL_FALSE;
203  if (iread(iCurImage->Pal.Palette, 1, 768) != 768)
204  return IL_FALSE;
205 
206  if (ilGetBoolean(IL_CONV_PAL) == IL_TRUE) {
207  ilConvertImage(IL_RGB, IL_UNSIGNED_BYTE);
208  }
209 
210  iseek(Position, IL_SEEK_SET);
211  }
212 
213  iCurImage = BaseImage;
214 
215  return ilFixImage();
216 }
