68 ILboolean iGetHdrHead(HDRHEADER *Header)
69 {
70  ILboolean done = IL_FALSE;
71  char a, b;
72  char x[3], y[3]; //changed 20050217: added space for the '\0' char
73  char buff[81]; // 01-19-2009: Added space for the '\0'.
74  ILuint count = 0;
75 
76  iread(Header->Signature, 1, 10);
77 
78  //skip lines until an empty line is found.
79  //this marks the end of header information,
80  //the next line contains the image's dimension.
81 
82  //TODO: read header contents into variables
83  //(EXPOSURE, orientation, xyz correction, ...)
84 
85  if (iread(&a, 1, 1) != 1)
86  return IL_FALSE;
87 
88  while (!done) {
89  if (iread(&b, 1, 1) != 1)
90  return IL_FALSE;
91  if (b == '\n' && a == '\n')
92  done = IL_TRUE;
93  else
94  a = b;
95  }
96 
97  //read dimensions (note that this assumes a somewhat valid image)
98  if (iread(&a, 1, 1) != 1)
99  return IL_FALSE;
100  while (a != '\n') {
101  if (count >= 80) // Line shouldn't be this long at all.
102  return IL_FALSE;
103  buff[count] = a;
104  if (iread(&a, 1, 1) != 1)
105  return IL_FALSE;
106  ++count;
107  }
108  buff[count] = '\0';
109 
110  //note that this is not the 100% correct way to load hdr images:
111  //in a perfect world we would check if there's a +/- X/Y in front
112  //of width and heigth and mirror + rotate the image after decoding
113  //according to that. But HDRShop doesn't do that either (and that's
114  //my reference program :) ) and it's just a rotate and a mirror,
115  //nothing that really changes the appearance of the loaded image...
116  //(The code as it is now assumes that y contains "-Y" and x contains
117  //"+X" after the following line)
118 
119  // The 2 has to be in the %s format specifier to prevent buffer overruns.
120  sscanf(buff, "%2s %d %2s %d", y, &Header->Height, x, &Header->Width);
121 
122  return IL_TRUE;
123 }

194 // Internal function used to load the .hdr.
195 ILboolean iLoadHdrInternal()
196 {
197  HDRHEADER Header;
198  ILfloat *data;
199  ILubyte *scanline;
200  ILuint i, j, e, r, g, b;
201 
202  if (iCurImage == NULL) {
203  ilSetError(IL_ILLEGAL_OPERATION);
204  return IL_FALSE;
205  }
206 
207  if (!iGetHdrHead(&Header)) {
208  ilSetError(IL_INVALID_FILE_HEADER);
209  return IL_FALSE;
210  }
211  if (!iCheckHdr(&Header)) {
212  //iseek(-(ILint)sizeof(HDRHEAD), IL_SEEK_CUR);
213  ilSetError(IL_INVALID_FILE_HEADER);
214  return IL_FALSE;
215  }
216 
217  // Update the current image with the new dimensions
218  if (!ilTexImage(Header.Width, Header.Height, 1, 3, IL_RGB, IL_FLOAT, NULL)) {
219  return IL_FALSE;
220  }
221  iCurImage->Origin = IL_ORIGIN_UPPER_LEFT;
222 
223  //read image data
224  if (iGetHint(IL_MEM_SPEED_HINT) == IL_FASTEST)
225  iPreCache(iCurImage->Width / 8 * iCurImage->Height);
226 
227  data = (ILfloat*)iCurImage->Data;
228  scanline = (ILubyte*)ialloc(Header.Width*4);
229  for (i = 0; i < Header.Height; ++i) {
230  ReadScanline(scanline, Header.Width);
231 
232  //convert hdrs internal format to floats
233  for (j = 0; j < 4*Header.Width; j += 4) {
234  ILuint *ee;
235  ILfloat t, *ff;
236  e = scanline[j + 3];
237  r = scanline[j + 0];
238  g = scanline[j + 1];
239  b = scanline[j + 2];
240 
241  //t = (float)pow(2.f, ((ILint)e) - 128);
242  if (e != 0)
243  e = (e - 1) << 23;
244 
245  // All this just to avoid stric-aliasing warnings...
246  // was: t = *(ILfloat*)&e
247  ee = &e;
248  ff = (ILfloat*)ee;
249  t = *ff;
250 
251  data[0] = (r/255.0f)*t;
252  data[1] = (g/255.0f)*t;
253  data[2] = (b/255.0f)*t;
254  data += 3;
255  }
256  }
257  iUnCache();
258  ifree(scanline);
259 
260  return ilFixImage();
261 }

263 void ReadScanline(ILubyte *scanline, ILuint w) {
264  ILubyte *runner;
265  ILuint r, g, b, e, read, shift;
266 
267  r = igetc();
268  g = igetc();
269  b = igetc();
270  e = igetc();
271 
272  //check if the scanline is in the new format
273  //if so, e, r, g, g are stored separated and are
274  //rle-compressed independently.
275  if (r == 2 && g == 2) {
276  ILuint length = (b << 8) | e;
277  ILuint j, t, k;
278  if (length > w)
279  length = w; //fix broken files
280  for (k = 0; k < 4; ++k) {
281  runner = scanline + k;
282  j = 0;
283  while (j < length) {
284  t = igetc();
285  if (t > 128) { //Run?
286  ILubyte val = igetc();
287  t &= 127;
288  //copy current byte
289  while (t > 0 && j < length) {
290  *runner = val;
291  runner += 4;
292  --t;
293  ++j;
294  }
295  }
296  else { //No Run.
297  //read new bytes
298  while (t > 0 && j < length) {
299  *runner = igetc();
300  runner += 4;
301  --t;
302  ++j;
303  }
304  }
305  }
306  }
307  return; //done decoding a scanline in separated format
308  }
309 
310  //if we come here, we are dealing with old-style scanlines
311  shift = 0;
312  read = 0;
313  runner = scanline;
314  while (read < w) {
315  if (read != 0) {
316  r = igetc();
317  g = igetc();
318  b = igetc();
319  e = igetc();
320  }
321 
322  //if all three mantissas are 1, then this is a rle
323  //count dword
324  if (r == 1 && g == 1 && b == 1) {
325  ILuint length = e;
326  ILuint j;
327  for (j = length << shift; j > 0 && read < w; --j) {
328  memcpy(runner, runner - 4, 4);
329  runner += 4;
330  ++read;
331  }
332  //if more than one rle count dword is read
333  //consecutively, they are higher order bytes
334  //of the first read value. shift keeps track of
335  //that.
336  shift += 8;
337  }
338  else {
339  runner[0] = r;
340  runner[1] = g;
341  runner[2] = b;
342  runner[3] = e;
343 
344  shift = 0;
345  runner += 4;
346  ++read;
347  }
348  }
349 }

414 /* standard conversion from float pixels to rgbe pixels */
415 /* note: you can remove the "inline"s if your compiler complains about it */
416 //static INLINE void
417 static void
418 float2rgbe(unsigned char rgbe[4], float red, float green, float blue)
419 {
420  float v;
421  int e;
422 
423  v = red;
424  if (green > v) v = green;
425  if (blue > v) v = blue;
426  if (v < 1e-32) {
427  rgbe[0] = rgbe[1] = rgbe[2] = rgbe[3] = 0;
428  }
429  else {
430  v = (float)(frexp(v,&e) * 256.0/v);
431  rgbe[0] = (unsigned char) (red * v);
432  rgbe[1] = (unsigned char) (green * v);
433  rgbe[2] = (unsigned char) (blue * v);
434  rgbe[3] = (unsigned char) (e + 128);
435  }
436 }

439 typedef struct {
440  ILuint valid; /* indicate which fields are valid */
441  ILbyte programtype[16]; /* listed at beginning of file to identify it

463 /* default minimal header. modify if you want more information in header */
464 ILboolean RGBE_WriteHeader(ILuint width, ILuint height, rgbe_header_info *info)
465 {
466  char *programtype = (char*)"RGBE";
467 
468  if (info && (info->valid & RGBE_VALID_PROGRAMTYPE))
469  programtype = (char*)info->programtype;
470  if (ilprintf("#?%s\n",programtype) < 0)
471  return IL_FALSE;
472  /* The #? is to identify file type, the programtype is optional. */
473  if (info && (info->valid & RGBE_VALID_GAMMA)) {
474  if (ilprintf("GAMMA=%g\n",info->gamma) < 0)
475  return IL_FALSE;
476  }
477  if (info && (info->valid & RGBE_VALID_EXPOSURE)) {
478  if (ilprintf("EXPOSURE=%g\n",info->exposure) < 0)
479  return IL_FALSE;
480  }
481  if (ilprintf("FORMAT=32-bit_rle_rgbe\n\n") < 0)
482  return IL_FALSE;
483  if (ilprintf("-Y %d +X %d\n", height, width) < 0)
484  return IL_FALSE;
485  return IL_TRUE;

506 /* The code below is only needed for the run-length encoded files. */
507 /* Run length encoding adds considerable complexity but does */
508 /* save some space. For each scanline, each channel (r,g,b,e) is */
509 /* encoded separately for better compression. */
510 
511 ILboolean RGBE_WriteBytes_RLE(ILubyte *data, ILuint numbytes)
512 {
513 #define MINRUNLENGTH 4
514  ILuint cur, beg_run, run_count, old_run_count, nonrun_count;
515  ILubyte buf[2];
516 
517  cur = 0;
518  while (cur < numbytes) {
519  beg_run = cur;
520  /* find next run of length at least 4 if one exists */
521  run_count = old_run_count = 0;
522  while((run_count < MINRUNLENGTH) && (beg_run < numbytes)) {
523  beg_run += run_count;
524  old_run_count = run_count;
525  run_count = 1;
526  // 01-25-2009: Moved test for beg_run + run_count first so that it is
527  // tested first. This keeps it from going out of bounds by 1.
528  while((beg_run + run_count < numbytes) && (run_count < 127) &&
529  (data[beg_run] == data[beg_run + run_count]))
530  run_count++;
531  }
532  /* if data before next big run is a short run then write it as such */
533  if ((old_run_count > 1)&&(old_run_count == beg_run - cur)) {
534  buf[0] = 128 + old_run_count; /*write short run*/
535  buf[1] = data[cur];
536  if (iwrite(buf,sizeof(buf[0])*2,1) < 1)
537  return IL_FALSE;
538  cur = beg_run;
539  }
540  /* write out bytes until we reach the start of the next run */
541  while(cur < beg_run) {
542  nonrun_count = beg_run - cur;
543  if (nonrun_count > 128)
544  nonrun_count = 128;
545  buf[0] = nonrun_count;
546  if (iwrite(buf,sizeof(buf[0]),1) < 1)
547  return IL_FALSE;
548  if (iwrite(&data[cur],sizeof(data[0])*nonrun_count,1) < 1)
549  return IL_FALSE;
550  cur += nonrun_count;
551  }
552  /* write out next run if one was found */
553  if (run_count >= MINRUNLENGTH) {
554  buf[0] = 128 + run_count;
555  buf[1] = data[beg_run];
556  if (iwrite(buf,sizeof(buf[0])*2,1) < 1)
557  return IL_FALSE;
558  cur += run_count;
559  }
560  }
561  return IL_TRUE;
562 #undef MINRUNLENGTH
563 }

566 // Internal function used to save the Hdr.
567 ILboolean iSaveHdrInternal()
568 {
569  ILimage *TempImage;
570  rgbe_header_info stHeader;
571  unsigned char rgbe[4];
572  ILubyte *buffer;
573  ILfloat *data;
574  ILuint i;
575  ILboolean bRet;
576 
577  if (iCurImage == NULL) {
578  ilSetError(IL_ILLEGAL_OPERATION);
579  return IL_FALSE;
580  }
581 
582  stHeader.exposure = 0;
583  stHeader.gamma = 0;
584  stHeader.programtype[0] = 0;
585  stHeader.valid = 0;
586 
587  if (iCurImage->Format != IL_UNSIGNED_BYTE) {
588  TempImage = iConvertImage(iCurImage, IL_RGB, IL_FLOAT);
589  if (TempImage == NULL)
590  return IL_FALSE;
591  }
592  else
593  TempImage = iCurImage;
594 
595  if (!RGBE_WriteHeader(TempImage->Width, TempImage->Height, &stHeader))
596  return IL_FALSE;
597 
598  if (TempImage->Origin == IL_ORIGIN_LOWER_LEFT)
599  iFlipBuffer(TempImage->Data, TempImage->Depth, TempImage->Bps, TempImage->Height);
600  data = (ILfloat*)TempImage->Data;
601 
602  if ((TempImage->Width < 8)||(TempImage->Width > 0x7fff)) {
603  /* run length encoding is not allowed so write flat*/
604  bRet = RGBE_WritePixels(data,TempImage->Width*TempImage->Height);
605  if (iCurImage != TempImage)
606  ilCloseImage(TempImage);
607  return bRet;
608  }
609  buffer = (ILubyte*)ialloc(sizeof(ILubyte)*4*TempImage->Width);
610  if (buffer == NULL) {
611  /* no buffer space so write flat */
612  bRet = RGBE_WritePixels(data,TempImage->Width*TempImage->Height);
613  if (iCurImage != TempImage)
614  ilCloseImage(TempImage);
615  return bRet;
616  }
617 
618  while(TempImage->Height-- > 0) {
619  rgbe[0] = 2;
620  rgbe[1] = 2;
621  rgbe[2] = TempImage->Width >> 8;
622  rgbe[3] = TempImage->Width & 0xFF;
623  if (iwrite(rgbe, sizeof(rgbe), 1) < 1) {
624  free(buffer);
625  if (iCurImage != TempImage)
626  ilCloseImage(TempImage);
627  return IL_FALSE;
628  }
629 
630  for(i=0;i<TempImage->Width;i++) {
631  float2rgbe(rgbe,data[RGBE_DATA_RED],data[RGBE_DATA_GREEN],data[RGBE_DATA_BLUE]);
632  buffer[i] = rgbe[0];
633  buffer[i+TempImage->Width] = rgbe[1];
634  buffer[i+2*TempImage->Width] = rgbe[2];
635  buffer[i+3*TempImage->Width] = rgbe[3];
636  data += RGBE_DATA_SIZE;
637  }
638  /* write out each of the four channels separately run length encoded */
639  /* first red, then green, then blue, then exponent */
640  for(i=0;i<4;i++) {
641  if (RGBE_WriteBytes_RLE(&buffer[i*TempImage->Width],TempImage->Width) != IL_TRUE) {
642  ifree(buffer);
643  if (iCurImage != TempImage)
644  ilCloseImage(TempImage);
645  return IL_FALSE;
646  }
647  }
648  }
649  ifree(buffer);
650 
651  if (iCurImage != TempImage)
652  ilCloseImage(TempImage);
653  return IL_TRUE;
654 }
