166 // Note: .Ktx support has not been tested yet!
167 // https://www.khronos.org/opengles/sdk/tools/KTX/file_format_spec/
168 ILboolean iLoadKtxInternal()
169 {
170  KTX_HEAD Header;
171  ILuint imageSize;
172  ILenum Format;
173  ILubyte Bpp;
174  char FileIdentifier[12] = {
175  //0xAB, 0x4B, 0x54, 0x58, 0x20, 0x31, 0x31, 0xBB, 0x0D, 0x0A, 0x1A, 0x0A
176  '«', 'K', 'T', 'X', ' ', '1', '1', '»', '\r', '\n', '\x1A', '\n'
177  };
178 
179  if (iCurImage == NULL) {
180  ilSetError(IL_ILLEGAL_OPERATION);
181  return IL_FALSE;
182  }
183 
184  /*Header.Width = GetLittleShort();
185  Header.Height = GetLittleShort();
186  Header.Dummy = GetLittleInt();*/
187 
188  if (iread(Header.identifier, 1, 12) != 12)
189  return IL_FALSE;
190  Header.endianness = GetLittleUInt();
191  Header.glType = GetLittleUInt();
192  Header.glTypeSize = GetLittleUInt();
193  Header.glFormat = GetLittleUInt();
194  Header.glInternalFormat = GetLittleUInt();
195  Header.glBaseInternalFormat = GetLittleUInt();
196  Header.pixelWidth = GetLittleUInt();
197  Header.pixelHeight = GetLittleUInt();
198  Header.pixelDepth = GetLittleUInt();
199  Header.numberOfArrayElements = GetLittleUInt();
200  Header.numberOfFaces = GetLittleUInt();
201  Header.numberOfMipmapLevels = GetLittleUInt();
202  Header.bytesOfKeyValueData = GetLittleUInt();
203 
204 
205  if (memcmp(Header.identifier, FileIdentifier, 12) || Header.endianness != 0x04030201) {
206  ilSetError(IL_ILLEGAL_FILE_VALUE);
207  return IL_FALSE;
208  }
209  //@TODO: Additional types
210  if (Header.glType != I_GL_UNSIGNED_BYTE || Header.glTypeSize != 1) {
211  ilSetError(IL_ILLEGAL_FILE_VALUE);
212  return IL_FALSE;
213  }
214  //@TODO: Additional formats
215  if (Header.glFormat <= I_GL_ALPHA || Header.glFormat >= I_GL_LUMINANCE_ALPHA || Header.glInternalFormat != Header.glFormat /*|| Header.glBaseInternalFormat != Header.glFormat*/) {
216  ilSetError(IL_ILLEGAL_FILE_VALUE);
217  return IL_FALSE;
218  }
219 
220  //@TODO: Mipmaps, etc.
221  if (Header.numberOfArrayElements != 0 || Header.numberOfFaces != 1 || Header.numberOfMipmapLevels != 1) {
222  ilSetError(IL_ILLEGAL_FILE_VALUE);
223  return IL_FALSE;
224  }
225  //@TODO: Parse this data
226  if (iseek(Header.bytesOfKeyValueData, IL_SEEK_CUR))
227  return IL_FALSE;
228 
229  switch (Header.glFormat)
230  {
231  case I_GL_LUMINANCE:
232  Bpp = 1;
233  Format = IL_LUMINANCE;
234  break;
235  case IL_LUMINANCE_ALPHA:
236  Bpp = 2;
237  Format = IL_LUMINANCE_ALPHA;
238  break;
239  case I_GL_RGB:
240  Bpp = 3;
241  Format = IL_RGB;
242  break;
243  case I_GL_RGBA:
244  Bpp = 4;
245  Format = IL_RGBA;
246  break;
247  default:
248  ilSetError(IL_ILLEGAL_FILE_VALUE);
249  return IL_FALSE;
250  }
251 
252  //@TODO: More than just RGBA
253  if (!ilTexImage(Header.pixelWidth, Header.pixelHeight, 1, Bpp, Format, IL_UNSIGNED_BYTE, NULL)) {
254  return IL_FALSE;
255  }
256  iCurImage->Origin = IL_ORIGIN_UPPER_LEFT;
257 
258  imageSize = GetLittleUInt();
259  if (imageSize != Header.pixelWidth * Header.pixelHeight * Bpp) {
260  ilSetError(IL_ILLEGAL_FILE_VALUE);
261  return IL_FALSE;
262  }
263 
264  if (iread(iCurImage->Data, Bpp, Header.pixelWidth * Header.pixelHeight) != Header.pixelWidth * Header.pixelHeight)
265  return IL_FALSE;
266 
267  return ilFixImage();
268 }
