27 typedef struct VTFHEAD
28 {
29  ILubyte Signature[4]; // File signature ("VTF\0").
30  ILuint Version[2]; // version[0].version[1] (currently 7.2).
31  ILuint HeaderSize; // Size of the header struct (16 byte aligned; currently 80 bytes).
32  ILushort Width; // Width of the largest mipmap in pixels. Must be a power of 2.
33  ILushort Height; // Height of the largest mipmap in pixels. Must be a power of 2.
34  ILuint Flags; // VTF flags.
35  ILushort Frames; // Number of frames, if animated (1 for no animation).
36  ILushort FirstFrame; // First frame in animation (0 based).
37  ILubyte Padding0[4]; // reflectivity padding (16 byte alignment).
38  ILfloat Reflectivity[3]; // reflectivity vector.
39  ILubyte Padding1[4]; // reflectivity padding (8 byte packing).
40  ILfloat BumpmapScale; // Bumpmap scale.
41  ILuint HighResImageFormat; // High resolution image format.
42  ILubyte MipmapCount; // Number of mipmaps.
43  ILuint LowResImageFormat; // Low resolution image format (always DXT1).
44  ILubyte LowResImageWidth; // Low resolution image width.
45  ILubyte LowResImageHeight; // Low resolution image height.
46  ILushort Depth; // Depth of the largest mipmap in pixels.
47  // Must be a power of 2. Can be 0 or 1 for a 2D texture (v7.2 only).
48 } IL_PACKSTRUCT VTFHEAD;
